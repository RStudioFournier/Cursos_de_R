EMVR <- function(dat, dist){
if(dist == "Normal" | dist == "normal" | dist == "n" | dist == "N"){
  mu <- mean(dat)
  var <- (sum((dat-mu)^2))/(length(dat))
  cat("Estimadores de maxima verosimilitud de los datos ingresados  
  con distribucion normal", "\n", "Media =", mu, "\n", "Varianza =",
  var) }
  
if(dist == "Binomial" | dist == "binomial" | dist == "b" | 
  dist == "B"){
  p <- (sum(dat))/(length(dat))
  cat("Estimador de maxima verosimilitud de los datos ingresados 
  con distribucion binomial", "\n", "p =", p) }
}
